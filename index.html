import tkinter as tk from tkinter import ttk # Bank data derived from Bank of Mongolia definitions # Format: "Display Name": "4-digit Bank Identifier used in IBAN" BANKS = { "Khan Bank": "0005", "Trade and Development Bank (TDB)": "0004", "Golomt Bank": "0015", "XacBank": "0032", "State Bank": "0034", "Bogd Bank": "0038", "M Bank": "0039", "Arig Bank": "0021", "TransBank": "0019", "Capitron Bank": "0030", "Chinggis Khaan Bank": "0033", "Development Bank of Mongolia": "0036", "Mongolbank (Central Bank)": "0001" } def calculate_iban_check_digits(bank_code, account_padded): """ Calculates the 2-digit MOD-97 check digits for a Mongolian IBAN. Structure for calculation: [BankCode][Account] + [M=22][N=23] + [00] """ # 2223 is the numeric representation of 'MN' (M=22, N=23) rearranged = f"{bank_code}{account_padded}222300" # MOD 97 logic: Check Digits = 98 - (Rearranged_Numeric % 97) remainder = int(rearranged) % 97 check_digits = 98 - remainder # Return as a zero-padded 2-digit string return f"{check_digits:02d}" def update_iban(*args): selected_bank = bank_var.get() account_input = account_var.get().strip() # If no bank is selected, clear everything if not selected_bank or selected_bank == "Select a Bank...": result_label.config(text="") info_label.config(text="Please select a financial institution.", fg="#7f8c8d") return bank_code = BANKS[selected_bank] # Strip any spaces or accidental non-numeric characters from the input clean_account = "".join(filter(str.isdigit, account_input)) if not clean_account: # Scenario A: Bank selected but no account number typed yet # Shows bank-specific structural information placeholder result_label.config(text="") info_label.config( text=f"Bank Identifier: {bank_code} | Awaiting account number...", fg="#d35400" ) else: # Pad account number to exactly 12 digits with leading zeros padded_account = clean_account.zfill(12) # Calculate secure ISO 13616 compliant check digits check_digits = calculate_iban_check_digits(bank_code, padded_account) # Assemble standard 20-character IBAN full_iban = f"MN{check_digits}{bank_code}{padded_account}" # Pretty-print formatting (Groups of 4 for financial legibility) formatted_iban = " ".join(full_iban[i:i+4] for i in range(0, len(full_iban), 4)) result_label.config(text=formatted_iban) info_label.config(text="✓ Valid Mongolian IBAN structure generated.", fg="#27ae60") # --- UI Setup --- root = tk.Tk() root.title("Bank of Mongolia — Interbank IBAN Utility") root.geometry("620x340") root.configure(bg="#0f1e36") # Corporate navy/midnight blue background root.resizable(False, False) # Custom Styling style = ttk.Style() style.theme_use('clam') style.configure("TCombobox", fieldbackground="#ffffff", background="#dcdde1", foreground="#2c3e50", font=("Helvetica", 11)) # Top Banner Header header_frame = tk.Frame(root, bg="#0a1526", height=60) header_frame.pack(fill="x", side="top") header_label = tk.Label( header_frame, text="MONGOLIAN IBAN GENERATION PORTAL", font=("Helvetica", 13, "bold"), fg="#f1c40f", # Gold text bg="#0a1526" ) header_label.pack(pady=15) # Main Container Form form_frame = tk.Frame(root, bg="#0f1e36", padx=30, pady=20) form_frame.pack(fill="both", expand=True) # Dropdown Selection lbl_bank = tk.Label(form_frame, text="Select Institution:", font=("Helvetica", 10, "bold"), fg="#ffffff", bg="#0f1e36") lbl_bank.grid(row=0, column=0, sticky="w", pady=(0, 5)) bank_var = tk.StringVar() bank_dropdown = ttk.Combobox(form_frame, textvariable=bank_var, width=32, state="readonly") bank_dropdown['values'] = ["Select a Bank..."] + list(BANKS.keys()) bank_dropdown.current(0) bank_dropdown.grid(row=1, column=0, padx=(0, 20), sticky="w") bank_var.trace_add("write", update_iban) # Account Input Textfield lbl_acc = tk.Label(form_frame, text="Pre-IBAN Account Number:", font=("Helvetica", 10, "bold"), fg="#ffffff", bg="#0f1e36") lbl_acc.grid(row=0, column=1, sticky="w", pady=(0, 5)) account_var = tk.StringVar() account_entry = tk.Entry(form_frame, textvariable=account_var, width=24, font=("Helvetica", 12), bg="#ffffff", fg="#2c3e50", bd=2, relief="groove") account_entry.grid(row=1, column=1, sticky="w") account_var.trace_add("write", update_iban) # Separator rule separator = tk.Frame(form_frame, height=2, bd=1, relief="sunken", bg="#1b3052") separator.grid(row=2, column=0, columnspan=2, sticky="ew", pady=25) # Dynamic Output Section lbl_output = tk.Label(form_frame, text="Official IBAN Registry Format:", font=("Helvetica", 9, "bold"), fg="#95a5a6", bg="#0f1e36") lbl_output.grid(row=3, column=0, columnspan=2, sticky="w") result_label = tk.Label( form_frame, text="", font=("Courier New", 20, "bold"), fg="#f1c40f", # Crisp gold for readability bg="#112544", relief="solid", bd=1, width=28, height=2 ) result_label.grid(row=4, column=0, columnspan=2, pady=(5, 5), sticky="ew") # Real-time status tracker info_label = tk.Label(form_frame, text="Please select a financial institution.", font=("Helvetica", 9, "italic"), fg="#7f8c8d", bg="#0f1e36") info_label.grid(row=5, column=0, columnspan=2, sticky="w", pady=(2, 0)) root.mainloop()
